```typescript
import { sessionState, getPhaseGuidance } from '../session/state.js';

export function registerSessionTools() {
  return [{
    name: "get_session_phase",
    description: "Get current brainstorm session phase (Double Diamond), engagement signals, and recommended approach",
    inputSchema: {
      type: "object" as const,
      properties: {},
      required: []
    }
  }];
}

export function handleSessionTool(name: string) {
  if (name === "get_session_phase") {
    const mode = (sessionState.phase === 'discover' || sessionState.phase === 'develop')
      ? 'diverge' : 'converge';
    return {
      content: [{
        type: "text" as const,
        text: JSON.stringify({
          phase: sessionState.phase,
          diamond: sessionState.diamond,
          mode,
          turnCount: sessionState.turnCount,
          currentTechnique: sessionState.currentTechnique,
          recommendedAction: getPhaseGuidance(sessionState)
        }, null, 2)
      }]
    };
  }
  return { content: [{ type: "text" as const, text: "Unknown session tool" }], isError: true };
}
```

**sessionStore.ts** - Follow preferencesStore.ts pattern:
```typescript
import { create } from 'zustand';

type DiamondPhase = 'discover' | 'define' | 'develop' | 'deliver';
type VisualizationTechnique = 'mindmap' | 'matrix' | 'affinity' | 'flow';

interface SessionStoreState {
  phase: DiamondPhase;
  diamond: 1 | 2;
  turnCount: number;
  technique: VisualizationTechnique;

  updateFromServer: (data: {
    phase: DiamondPhase;
    diamond: 1 | 2;
    turnCount: number;
    currentTechnique: VisualizationTechnique;
  }) => void;
}

export const useSessionStore = create<SessionStoreState>()((set) => ({
  phase: 'discover',
  diamond: 1,
  turnCount: 0,
  technique: 'mindmap',

  updateFromServer: (data) => set({
    phase: data.phase,
    diamond: data.diamond,
    turnCount: data.turnCount,
    technique: data.currentTechnique
  })
}));
```
