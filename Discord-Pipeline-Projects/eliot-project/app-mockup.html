  • /api/search — fuzzy name matching
  • /api/export — CSV download endpoint

// lib/data/cache.ts — CSV cache singleton import Papa from 'papaparse'; import fs from 'fs'; class DataCache { private cache = new Map<string, any[]>(); async loadAll() { const files = [ 'egg_db', 'growth_db', 'settlement_age_db', 'settlement_size_db', 'vertical_position_db', 'pelagic_juvenile_db', 'rafting_db', 'swimming_ucrit_db', 'swimming_iss_db' ]; for (const f of files) { const csv = fs.readFileSync(\`database/\${f}.csv\`, 'utf-8'); this.cache.set(f, Papa.parse(csv, { header: true }).data); } } } // pages/species/[name].tsx — ISR config export async function getStaticProps({ params }) { const data = await cache.getSpecies(params.name); return { props: { data }, revalidate: 3600 }; }
